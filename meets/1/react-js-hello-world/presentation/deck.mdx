# Arrow Notation


```jsx

const HelloWorld = function () { 
	return <p>Hi, I'm Groot!</p>
}

const HelloWorld = () => {
	return <p>Hi, I'm Groot!</p>
}

const HelloWorld = () => <p>Hi, I'm Groot!</p>

```

```jsx
<HelloWorld />
```

# Properties

```

	return <p>Hi, I'm {props.name}!</p>
}

({name}) => <p>Hi, I'm {name}!</p>
```

# Render


```jsx

import ReactDOM from 'react-dom';

const HelloWorld = ({name}) => <p>Hi, I'm {name}!</p>

ReactDOM.render(<HelloWorld name="froso"/>,
 document.getElementById('root'));

```

```html
    style={{ padding: '20px', backgroundColor: 'tomato' }
    id='root'
```
