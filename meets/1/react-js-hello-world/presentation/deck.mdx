```jsx

  }

  render() {
  	   return <p>Hi, I'm Groot!</p>
  }

}

```

# Arrow Notation


```jsx

const HelloWorld = function () { 
	return <p>Hi, I'm Groot!</p>
}

const HelloWorld = () => {
	return <p>Hi, I'm Groot!</p>
}

const HelloWorld = () => <p>Hi, I'm Groot!</p>

```

```jsx
<HelloWorld />
```
